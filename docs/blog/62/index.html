projectile -- project management (e.g., find in project, run terminal, etc.)

perspective -- workspace manager (e.g., separate Emacs buffers for different projects)

vterm -- haven't used this in a while, but then we don't need an extra terminal to run simple things

(defun find-in-subtree (filename)
  (interactive "sFilename: ")
  (find-name-dired "" filename))

          (lambda ()
            (local-set-key (kbd "f") 'find-in-subtree)

;;; Kill the other window easily
(defun kill-buffer-in-other-window ()
  (interactive)
  "I just find this handy sometimes when working in two windows."
  (other-window 1)
  (kill-buffer)
  (other-window -1))
(global-set-key (kbd "C-x c") 'kill-buffer-in-other-window)
